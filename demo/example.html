  2. 无配置动态示例

    利用行选择事件(tp.row-selected)来获取表格选中的项

    更多事件函数见文档

    1. <!-- html -->
    2. <!-- a container with classname table-popover fade -->
    3. <div id="example2" class="table-popover fade"></div>
    4. <!-- js -->
    5. <script>
    6.   const columns = [{data: "name",title: "名字"},{data: "gender",title: "性别"}];
    7.   const data = [{name: "张三",gender: "男"},{name: "西施",gender: "女"}];
    9.     $("#example2").tablePopover({
    10.             columns: columns,
    11.             data: data
    12.     });
    14.     $('#example2').on('tp.row-selected', function(e, rowData) {
    15.         $('#example2-selected').
    16.         html("选择的是：<br>姓名：" + rowData.name + '，性别：' + rowData.gender);
    17.     });
    18. </script>
